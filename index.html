# --- Python Script Runner ---
runpythonscript() {
    script="$1"
    shebang=$(head -n 1 "$script" 2>/dev/null)

    if echo "$shebang" | grep -q "python2"; then
        echo "[+] Detected python2"
        command -v python2 >/dev/null && python2 "$script" || echo "[!] python2 not available"
    elif echo "$shebang" | grep -q "python3"; then
        echo "[+] Detected python3"
        python3 "$script"
    else
        echo "[+] Defaulting to python"
        if command -v python >/dev/null; then
            python "$script"
        else
            python3 "$script"
        fi
    fi
}
